import numpy as np

def simulate(theta=0.3, W=1.0, mu=0.8, steps=200):
    t = np.linspace(0, 10, steps)
    phase = np.pi * theta / 4.0
    energy = W * np.sin(t + phase)          # E：能量波

    return t, energy, matter
